add_namespace = tesb_ITA

country_event = {
	id = tesb_ITA.1
	title = tesb_ITA.1.t
	desc = tesb_ITA.1.d
	picture = GFX_report_event_royal_parade

	is_triggered_only = yes
	
	option = {
		name = tesb_ITA.1.a
		ITA = {
			country_event = { id = tesb_ITA.2 }
			if = {
				limit = {
					NOT = { has_opinion_modifier = ITA_crowned_king }
				}
				add_opinion_modifier = {
					target = ROOT
					modifier = ITA_crowned_king
				}
			}
		}
		ai_chance = {
			base = 5
			modifier = {
				factor = 0.1
				NOT = { has_government = absolutist }
			}
			modifier = {
				factor = 0
				has_war_with = ITA
			}
			modifier = {
				factor = 1.5
				has_opinion = {
					target = ITA
					value > 24
				}
			}
			modifier = {
				factor = 2
				has_opinion = {
					target = ITA
					value > 49
				}
			}
		}
	}

	option = {
		name = tesb_ITA.1.b
		ITA = {
			country_event = { id = tesb_ITA.3 }
			if = {
				limit = {
					NOT = { has_opinion_modifier = ITA_refused_crown }
				}
				add_opinion_modifier = {
					target = ROOT
					modifier = ITA_refused_crown
				}
			}
		}
		ai_chance = {
			base = 1
			modifier = {
				factor = 1.5
				has_opinion = {
					target = ITA
					value < 0
				}
			}
			modifier = {
				factor = 2
				has_opinion = {
					target = ITA
					value < -25
				}
			}
			modifier = {
				factor = 5
				has_opinion = {
					target = ITA
					value < -50
				}
			}
		}
	}
}

country_event = {
	id = tesb_ITA.10
	title = tesb_ITA.10.t
	desc = tesb_ITA.10.desc
	picture = GFX_report_event_generic_sign_treaty1

	is_triggered_only = yes

	option = {
		trigger = {
			is_faction_leader = no
		}
		name = tesb_ITA.10.a
		set_country_flag = join_ITA
		ITA = {
			country_event = { id = tesb_ITA.11 hours = 24 }
		}
		ai_chance = {
			base = 10
		}
	}
	option = {
		trigger = {
			is_faction_leader = yes
		}
		name = tesb_ITA.10.b
		set_country_flag = join_ITA
		ITA = {
			country_event = { id = tesb_ITA.11 hours = 24 }
		}
		every_country = {
			limit = {
				is_in_faction_with = ROOT
			}
			country_event = { id = tesb_ITA.12 }
		}
		ai_chance = {
			base = 10
		}
	}
	option = {
		name = tesb_ITA.10.c
		set_country_flag = reject_ITA
		ITA = {
			country_event = { id = tesb_ITA.13 hours = 24 }
			effect_tooltip = {
				create_wargoal = {
					target = ROOT
					type = puppet_wargoal_focus
				}
			}
		}
		ai_chance = {
			base = 1
		}
	}
}

	}
