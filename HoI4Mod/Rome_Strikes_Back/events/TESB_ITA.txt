country_event = {
	id = tesb_ITA.1
	title = tesb_ITA.1.t
	desc = tesb_ITA.1.d
	picture = GFX_report_event_royal_parade

	is_triggered_only = yes
	
	option = {
		name = tesb_ITA.1.a
		ITA = {
			country_event = { id = tesb_ITA.2 }
			if = {
				limit = {
					NOT = { has_opinion_modifier = ITA_crowned_king }
				}
				add_opinion_modifier = {
					target = ROOT
					modifier = ITA_crowned_king
				}
			}
		}
		ai_chance = {
			base = 5
			modifier = {
				factor = 0.1
				NOT = { has_government = absolutist }
			}
			modifier = {
				factor = 0
				has_war_with = ITA
			}
			modifier = {
				factor = 1.5
				has_opinion = {
					target = ITA
					value > 24
				}
			}
			modifier = {
				factor = 2
				has_opinion = {
					target = ITA
					value > 49
				}
			}
		}
	}

	option = {
		name = tesb_ITA.1.b
		ITA = {
			country_event = { id = tesb_ITA.3 }
			if = {
				limit = {
					NOT = { has_opinion_modifier = ITA_refused_crown }
				}
				add_opinion_modifier = {
					target = ROOT
					modifier = ITA_refused_crown
				}
			}
		}
		ai_chance = {
			base = 1
			modifier = {
				factor = 1.5
				has_opinion = {
					target = ITA
					value < 0
				}
			}
			modifier = {
				factor = 2
				has_opinion = {
					target = ITA
					value < -25
				}
			}
			modifier = {
				factor = 5
				has_opinion = {
					target = ITA
					value < -50
				}
			}
		}
	}
}
