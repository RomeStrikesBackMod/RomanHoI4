				}
			}

				factor = 2
				has_opinion = {
					target = ITA

			}
		}

					target = ROOT
					modifier = ITA_refused_crown
				}
			}
		}
		ai_chance = {
			base = 1
			modifier = {
				factor = 1.5
				has_opinion = {
					target = ITA
					value < 0
				}
			}
