				if = { limit = { NOT = { has_character = JAP_myoukougein } } recruit_character = JAP_myoukougein }
				if = { limit = { NOT = { has_character = JAP_fujiwara_takeru } } recruit_character = JAP_fujiwara_takeru }
				if = { limit = { NOT = { has_character = JAP_minamoto_shigemori } } recruit_character = JAP_minamoto_shigemori }
				if = { limit = { NOT = { has_character = JAP_minamoto_bennosuke } } recruit_character = JAP_minamoto_bennosuke }
				if = { limit = { NOT = { has_character = JAP_iwamoto_kanehira } } recruit_character = JAP_iwamoto_kanehira }
				if = { limit = { NOT = { has_character = JAP_taira_yoritomo } } recruit_character = JAP_taira_yoritomo }
				if = { limit = { NOT = { has_character = JAP_fujiwara_tomoe } } recruit_character = JAP_fujiwara_tomoe }
				if = { limit = { NOT = { has_character = JAP_saitou_kichinosuke } } recruit_character = JAP_saitou_kichinosuke }
				if = { limit = { NOT = { has_character = JAP_sakamoto_hajime } } recruit_character = JAP_sakamoto_hajime }
				random_list = {
					5 = { set_country_flag = JAP_mutsuhito }
					95 = { set_country_flag = JAP_myoukougein  }
				}
				news_event = { id = tesb_JAP.2 days = 7 }

		name = tesb_JAP.1.a
	}
}
