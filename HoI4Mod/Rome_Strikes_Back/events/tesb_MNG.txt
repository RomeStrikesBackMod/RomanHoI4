### Invite Korea to faction events
# Invite Korea to faction
country_event = {
	id = tesb_MNG.1
	title = tesb_MNG.1.t
	desc = tesb_MNG.1.d
	picture = GFX_report_event_ger_civil_war

	is_triggered_only = yes
	# accept unconditionally
	option = {
		name = tesb_MNG.1.a
		MNG = { country_event = { id = tesb_MNG.2 } }
		trigger = {
			NOT = { has_war_with = MNG }
			is_in_faction = no
		}
		ai_chance = {
			base = 100
		}
	}
	# refuse
	option = {
		name = tesb_MNG.1.b
		MNG = { country_event = { id = tesb_MNG.3 } }
		ai_chance = {
			base = 0
		}
	}
}

### Nogotiate with Russia for Mongolia events
# Begin negotiations
country_event = {
	id = tesb_MNG.20
	title = tesb_MNG.20.t
	desc = tesb_MNG.20.d
	picture = GFX_report_event_ger_civil_war

	is_triggered_only = yes
	# accept unconditionally
	option = {
		name = tesb_MNG.20.a
		MNG = { country_event = { id = tesb_MNG.21 } }
		trigger = {
			NOT = { has_war_with = MNG }
			is_in_faction = no
		}
		ai_chance = {
			base = 100
		}
	}
	# refuse
	option = {
		name = tesb_MNG.20.b
		MNG = { country_event = { id = tesb_MNG.22 } }
		ai_chance = {
			base = 0
		}
	}
	# negotiate
	option = {
		name = tesb_MNG.20.c
		trigger = {
			NOT = { has_war_with = MNG }
			MNG = {
				owns_state = 337
			}
		}
		MNG = { country_event = { id = tesb_MNG.23 } }
		ai_chance = {
			base = 0
		}
	}
}

# Russia accepts
country_event = {
	id = tesb_MNG.21
	title = tesb_MNG.21.t
	desc = tesb_MNG.21.d
	picture = GFX_report_event_ger_civil_war

	is_triggered_only = yes
	option = {
		name = tesb_MNG.21.a
		MNG = {
			transfer_state = 510
		}
	}
}

# Russia refuses
country_event = {
	id = tesb_MNG.22
	title = tesb_MNG.22.t
	desc = tesb_MNG.22.d
	picture = GFX_report_event_ger_civil_war

	is_triggered_only = yes
	# allow them to refuse
	option = {
		name = tesb_MNG.22.a
		ai_chance = {
			base = 0
		}
	}
	# force them to give up Mongolia
	option = {
		name = tesb_MNG.22.b
		create_wargoal = {
			type = take_state
			target = RUS
			generator = { 510 }
		}
		ai_chance = {
			base = 1
		}
	}
}

# Russia negotiates
country_event = {
	id = tesb_MNG.23
	title = tesb_MNG.23.t
	desc = tesb_MNG.23.d
	picture = GFX_report_event_ger_civil_war

	is_triggered_only = yes
	# accept
	option = {
		name = tesb_MNG.23.a
		MNG = {
			transfer_state = 510
		}
		RUS = {
			transfer_state = 337
		}
	}
	# refuse
	option = {
		name = tesb_MNG.23.b
		RUS = { country_event = { id = tesb_MNG.24 } }
	}
}
