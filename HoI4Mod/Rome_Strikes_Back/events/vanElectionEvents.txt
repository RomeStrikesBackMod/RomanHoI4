# Absolutists in Government?
country_event = {
	id = van_election.3
	title = van_election.3.t
	desc = van_election.3.d
	picture = GFX_report_event_fascist_gathering
	
	is_triggered_only = yes

	trigger = {
		OR = {
			has_government = democratic
			has_government = neutrality
		}
		NOT = { has_idea = absolutist_defeated }
		absolutist > 0.15
	}
	
	option = {
		name = van_election.3.a
		ai_chance = {
			base = 0
			modifier = {
				add = 2
				can_lose_democracy_support = yes
			}
			modifier = {
				add = 1
				can_lose_unity = no
			}
		}
		add_popularity = {
			ideology = absolutist
			popularity = 0.1
		}
		set_country_flag = coalition_with_absolutists
	}

	option = {
		name = van_election.3.b
		ai_chance = {
			base = 0
			modifier = {
				add = 10
				can_lose_unity = yes
			}
		}
		add_stability = -0.05
		add_popularity = {
			ideology = democratic
			popularity = 0.03
		}
		set_country_flag = government_declined_absolutists
	}
}
